# 2. bcrypt
``` Javascript
const bcrypt = require('bcrypt');

userSchema.pre('save', async function(next){ // before saving to the database, the function will run first
    if(this.isModified('password')){
        const hash = await bcrypt.hash(this.password, 10);
        this.password = hash
    }
    next();
})
```
